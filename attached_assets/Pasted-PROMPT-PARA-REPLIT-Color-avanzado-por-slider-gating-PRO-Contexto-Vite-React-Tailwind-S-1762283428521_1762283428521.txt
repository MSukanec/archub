export type PlanCode = 'free' | 'pro' | 'teams';

export function isProOrTeams(plan?: PlanCode | string | null) {
  if (!plan) return false;
  const p = String(plan).toLowerCase();
  return p === 'pro' || p === 'teams';
}

4) Utilidades de color (H → hex, paleta)

Crea src/utils/color.ts:

export function hslToHex(h: number, s = 0.78, l = 0.55) {
  // s/l fijos para un look consistente tipo “Material You”
  const c = (1 - Math.abs(2 * l - 1)) * s;
  const x = c * (1 - Math.abs(((h / 60) % 2) - 1));
  const m = l - c/2;
  let r=0,g=0,b=0;

  if (0 <= h && h < 60)   { r=c; g=x; b=0; }
  else if (60 <= h && h < 120) { r=x; g=c; b=0; }
  else if (120 <= h && h < 180){ r=0; g=c; b=x; }
  else if (180 <= h && h < 240){ r=0; g=x; b=c; }
  else if (240 <= h && h < 300){ r=x; g=0; b=c; }
  else if (300 <= h && h <= 360){ r=c; g=0; b=x; }

  const toHex = (v:number) => {
    const h = Math.round((v + m) * 255).toString(16).padStart(2, '0');
    return h;
  };
  return `#${toHex(r)}${toHex(g)}${toHex(b)}`;
}

5) Componente slider (con gating por plan)

Crea src/components/projects/ProjectColorAdvanced.tsx:

import { useEffect, useMemo, useState } from "react";
import { hslToHex } from "@/utils/color";
import { useThemePreviewStore } from "@/store/themePreviewStore";
import { isProOrTeams } from "@/utils/plan";

type Props = {
  initialHue?: number | null;          // projects.custom_color_h
  initialEnabled?: boolean;            // projects.use_custom_color
  planCode?: string | null;            // 'free' | 'pro' | 'teams'
  disabledReason?: string;             // texto de bloqueo si no es PRO
  onChange: (p: { useCustom: boolean; hue: number | null; hex?: string }) => void;
};

export default function ProjectColorAdvanced({
  initialHue = null,
  initialEnabled = false,
  planCode,
  disabledReason = "Función disponible en planes PRO o TEAMS",
  onChange
}: Props) {
  const isPro = isProOrTeams(planCode);
  const [enabled, setEnabled] = useState(initialEnabled && isPro);
  const [h, setH] = useState<number>(initialHue ?? 0);
  const { setProjectAccentPreview } = useThemePreviewStore();

  const hex = useMemo(() => hslToHex(h), [h]);

  useEffect(() => {
    if (enabled && isPro) setProjectAccentPreview(hex);
    else setProjectAccentPreview(undefined);
    return () => setProjectAccentPreview(undefined);
  }, [enabled, isPro, hex, setProjectAccentPreview]);

  useEffect(() => {
    onChange({ useCustom: enabled && isPro, hue: enabled && isPro ? h : null, hex: enabled && isPro ? hex : undefined });
  }, [enabled, h, isPro]);

  return (
    <div className="rounded-xl border p-3 mt-3 relative">
      <div className="flex items-center justify-between mb-2">
        <div className="font-medium">Color avanzado</div>
        <label className="flex items-center gap-2 text-sm">
          <input
            type="checkbox"
            className="accent-current"
            checked={enabled && isPro}
            onChange={(e) => setEnabled(e.target.checked)}
            disabled={!isPro}
          />
          Usar slider
        </label>
      </div>

      {!isPro && (
        <div className="absolute inset-0 bg-black/5 backdrop-blur-[2px] rounded-xl flex items-center justify-center text-sm font-medium">
          {disabledReason}
        </div>
      )}

      <div className="flex items-center gap-4">
        <input
          type="range"
          min={0}
          max={360}
          value={h}
          onChange={(e) => setH(Number(e.target.value))}
          className="w-full"
          disabled={!isPro || !enabled}
        />
        <div
          className="w-10 h-10 rounded-full ring-2 ring-black/10 shrink-0"
          style={{ background: hex }}
          title={hex}
        />
      </div>

      <div className="mt-2 text-xs text-black/60">
        Tono: {h}° · {hex}
      </div>
    </div>
  );
}
